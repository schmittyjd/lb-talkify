What browserified code looks like


          var $ = require('jquery');
          var _ = require('underscore');

          $('.stuff').addClass('moreStuff');
          var dude = {
            aka: 'jeff bridges',
            favDrink: 'white russian'
          };

          _.extend(dude, {
            nickname: 'the dude'
          });

          console.log(dude.nickname); // the dude

A look at basic commonjs

If exporting a singleton function or object use module.exports


          // sum.js

          module.exports = function(x, y) {
            return x + y;
          };

          // app.js

          var sum = require('sum.js');

          sum(3, 2); // 5

Named exports

If you are exporting multiple functions or objects use named exports with export.exportnamehere


        // simplemath.js
        exports.square = function(x) {
          return x * x;
        };

        exports.area = function(l, w) {
          return l * w;
        };

        // app.js
        var math = require('simplemath.js');

        math.square(2); // 4
        math.area(3, 5); // 15
